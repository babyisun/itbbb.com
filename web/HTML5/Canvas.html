/***直线***/
self.canvas.lineWidth = 10; //线宽度
self.canvas.strokeStyle = "red"; //线颜色
									
self.canvas.lineCap="butt";//默认。向线条的每个末端添加平直的边缘。
self.canvas.beginPath();
self.canvas.moveTo(100, 50); //起点坐标(x,y)
self.canvas.lineTo(250, 50); //终点坐标(x,y)
self.canvas.stroke();
									
self.canvas.lineCap="round";//向线条的每个末端添加圆形线帽。
self.canvas.strokeStyle="yellow";
self.canvas.beginPath();
self.canvas.moveTo(100, 100); //起点坐标(x,y)
self.canvas.lineTo(250, 100); //终点坐标(x,y)
self.canvas.stroke();
									
self.canvas.lineCap="square";//向线条的每个末端添加正方形线帽。
self.canvas.strokeStyle="#ccc";
self.canvas.beginPath();
self.canvas.moveTo(100, 150); //起点坐标(x,y)
self.canvas.lineTo(250, 150); //终点坐标(x,y)
self.canvas.stroke();
/***矩形***/
self.canvas.lineWidth = 3; //线宽度
self.canvas.strokeStyle = "#000"; //线颜色
self.canvas.beginPath();
self.canvas.strokeRect(130, 30, 170, 140); //空心矩形1(x,y,width,height)
//self.canvas.rect(30, 30,170, 140); //等同于矩形1(x,y,width,height)								
//self.canvas.stroke();
self.canvas.fillRect(130, 200, 170, 50); //实心矩形2(x,y,width,height)
//self.canvas.rect(130,200,170,50);//等同于矩形2
//self.canvas.fill();
self.canvas.stroke();
/***圆形***/
self.canvas.lineWidth = 2; //线宽度
self.canvas.strokeStyle = "#ccc"; //线颜色
self.canvas.beginPath();
//arc(x,y,r,sAngle,eAngle,counterclockwise)
//x	圆的中心的 x 坐标。
//y	圆的中心的 y 坐标。
//r	圆的半径。
//sAngle	起始角，以弧度计。（弧的圆形的三点钟位置是 0 度）。
//eAngle	结束角，以弧度计。
//counterclockwise	可选。规定应该逆时针还是顺时针绘图。False = 顺时针，true = 逆时针								
self.canvas.arc(200, 150, 100, 0, Math.PI * 2);
self.canvas.stroke(); //空心圆

//4分之3圆
self.canvas.beginPath();
self.canvas.arc(200, 150, 80, 0, Math.PI * 1.5);
self.canvas.stroke();

//实心圆
self.canvas.fillStyle = "yellow"; //填充颜色
self.canvas.lineWidth = 10; //线宽度
self.canvas.strokeStyle = "#ccc"; //线颜色
self.canvas.beginPath();
self.canvas.arc(200, 150, 50, 0, 90 * Math.PI / 180, true); //逆时针
self.canvas.stroke();//画线
self.canvas.fill();//填充	

/***圆角***/
self.canvas.beginPath();
self.canvas.strokeStyle = "red";
self.canvas.lineWidth = 2;
self.canvas.moveTo(60, 40); //起始点(x,y)
self.canvas.lineTo(140, 40); //移动到点(x,y)
//弧线(x1,y1,x2,y2,radius)，见图
self.canvas.arcTo(180, 40, 180, 80, 40); 
self.canvas.lineTo(180, 160);
self.canvas.stroke();
//辅助线
self.canvas.beginPath();
self.canvas.strokeStyle = "#ccc";
self.canvas.lineWidth = 1;
self.canvas.moveTo(140, 40);
self.canvas.lineTo(180, 40);
self.canvas.lineTo(180, 80);
self.canvas.stroke();
//说明文字
self.canvas.font = "18px 黑体";
self.canvas.fillStyle = "#ccc";
self.canvas.fillText("x1,y1", 185, 40); //fillText(text,x,y,[maxWidth]):
self.canvas.fillText("x2,y2", 185, 80);	

self.canvas.beginPath();
self.canvas.moveTo(100, 100);//起始点
//二次贝塞尔曲线
//quadraticCurveTo(cpx,cpy,x,y)
//cpx,cpy控制点1坐标
//x,y控制点2坐标
self.canvas.quadraticCurveTo(20, 50, 200, 20);
self.canvas.stroke();
self.canvas.beginPath();
self.canvas.moveTo(100, 250);//起始点
//三次贝塞尔曲线
//bezierCurveTo(cpx1,cpy1,cpx2,cpy2,endx,endy)
//cpx1,cpy1控制点1坐标
//cpx2,cpy2控制点2坐标
//endx,endy结束点坐标
self.canvas.bezierCurveTo(20, 150, 120, 150, 200, 180);
self.canvas.stroke();

//辅助点
self.canvas.beginPath();
self.canvas.fillStyle = "orange";
self.canvas.arc(100, 100, 3, 0, 2 * Math.PI);//起始点
self.canvas.arc(20, 50, 3, 0, 2 * Math.PI);//cpx,cpy
self.canvas.arc(200, 20, 3, 0, 2 * Math.PI);//x,y
self.canvas.arc(100, 250, 3, 0, 2 * Math.PI);//起始点2
self.canvas.arc(20, 150, 3, 0, 2 * Math.PI);//cpx1,cpy1
self.canvas.arc(120, 150, 3, 0, 2 * Math.PI);//cpx2,cpy2
self.canvas.arc(200, 180, 3, 0, 2 * Math.PI);//endx,endy
self.canvas.fill();

//说明文字
self.canvas.font = "18px黑体";
self.canvas.fillStyle = "#ccc";
self.canvas.fillText("cpx,cpy", 20, 45);//fillText(text,x,y,[maxWidth])
self.canvas.fillText("x,y", 200, 15);
self.canvas.fillText("起始点", 105, 250);
//起始点2
self.canvas.fillText("cpx1,cpy1", 25, 150);//cpx1,cpy1
self.canvas.fillText("cpx2,cpy2", 125, 150);//cpx2,cpy2
self.canvas.fillText("endx,endy", 205, 180);
